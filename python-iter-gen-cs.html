Python Cheat Sheet

Cheat sheet of Python. Some basic iterator and generator concept for Python programmer need to know.

More Readable Iteration in Python

>>> l = [1,2,3,4,5]
>>> i=0
>>> while i<len(l):
...   print l[i],
...   i+=1
... 
1 2 3 4 5
# more readable in python
>>> for _idx in range(len(l)):
...   print l[_idx],
... 
1 2 3 4 5
# best
>>> for _item in l:
...   print _item,
... 
1 2 3 4 5

Iterate list - Get List Item

>>> my_list = [1,2,3,4,5]
>>> for _i in my_list:
...   print _i,
... 
1 2 3 4 5

Iterate string - Get List Item

>>> my_str = "Hello"
>>> for _s in my_str:
...   print _s,
... 
H e l l o

Iterate dict - Get Keys

>>> my_dict={'k1':1,'k2':2,'k3':3}
>>> for _d in my_dict:
...   print _d,
... 
k3 k2 k1

Iterate dictionary - Get Values

>>> my_dict.values()
[3, 2, 1]
>>> for _v in my_dict.values():
...   print _v,
... 
3 2 1

#other way
>>> for _v in my_dict.itervalues():
...   print _v,
... 
3 2 1

Iterate dictionary - Get Keys and Values at The Same Time

>>> my_dict={'k1':1,'k2':2,'k3':3}
>>> my_dict.items()
[('k3', 3), ('k2', 2), ('k1', 1)]
>>> for _k,_v in my_dict.items():
...   print _k, _v
... 
k3 3
k2 2
k1 1

# other way
>>> for _k,_v in my_dict.iteritems():
...   print _k,_v 
... 
k3 3
k2 2
k1 1

Example of Use of Iterable In Python

# list, dict are example of "iterable" obj
# example 1: copy a list
>>> l = [1,2,3,4,5]
>>> new_l = list(l)
>>> new_l
[1, 2, 3, 4, 5]

# example 2: doing like map
>>> f= lambda x:x**2
>>> res=[f(x) for x in l]
>>> res
[1, 4, 9, 16, 25]

# example 3: calculate mean
>>> sum(l)/float(len(l))
3.0

# example 4: join string
>>> l = ['h','e','l','l','o']
>>> "".join(l)
'hello'

How do I Get Index at the same time?

>>> l = [1,2,3]
# not a good way in Python
>>> for _i in range(len(l)):
...   print _i, l[_i]
... 
0 1
1 2
2 3
# better way
>>> for _i,_v in enumerate(l):
...   print _i, _v
... 
0 1
1 2
2 3
# enumerate() makes useful pairs
>>> a = ['yo','~','man']
>>> list(enumerate(a))
[(0, 'yo'), (1, '~'), (2, 'man')]

How do I Loop Over Two lists?

# not a good way in Python
>>> names=['bob','alice']
>>> weights=[60,55]
>>> for _i in range(len(names)):
...   name = names[_i]
...   weight = weights[_i]
...   print name, weight
... 
bob 60
alice 55

# better way - using zip()
>>> for _n,_w in zip(names,weights):
...   print _n,_w
... 
bob 60
alice 55

dict() Accept a stream of pairs

# transfer between pair list and dict
>>> d={'k3': 3, 'k2': 2, 'k1': 1}
>>> pairs=d.items()
>>> pairs
[('k3', 3), ('k2', 2), ('k1', 1)]
>>> new_d = dict(pairs)
>> new_d
{'k3': 3, 'k2': 2, 'k1': 1}

# other example - zip two list
>>> names=['bob','alice']
>>> weights=[60,55]
>>> dict(zip(name,weights))
{'a': 60, 'l': 55}

Powerful of pair and dict

# find max value
>>> max(d.values())
3

# find max key and value via value
>>> cmp = lambda x: x[1]
>>> max(d.items(),key=cmp)
('k3', 3)

# find the 'key' which has max value
>>> d = {'1':3,'2':2,'3':0}
>>> max(d, key=d.get)
'1'
