Relationship between generator and contextmanager

>>> import contextlib
>>> @contextlib.contextmanager
... def mylist():
...   try:
...     l = [1,2,3,4,5]
...     yield l
...   finally:
...     print("exit scope")
... 
>>> with mylist() as l:
...   print(l)
... 
[1, 2, 3, 4, 5]
exit scope
